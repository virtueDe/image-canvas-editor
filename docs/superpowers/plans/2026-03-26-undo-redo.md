# Undo / Redo Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** 为现有图片编辑器增加同会话内的撤销与重做能力，覆盖已提交编辑结果，并保持草稿、裁剪、导出与现有 UI 行为兼容。

**Architecture:** 历史逻辑放在 `editor-core`，采用“结果快照 + undo/redo 双栈 + 图片注册表”的方案。快照只保存可重建的编辑结果，不保存 `cropMode`、`draftCropRect`、`viewport` 等临时交互状态；`editor-vue` 只桥接 `undo/redo` 与滑杆预览/提交；`web-vue` 只接最小按钮交互。

**Tech Stack:** TypeScript、Vue 3、Vite、UnoCSS、Canvas 2D、Vitest（新增）、`pnpm --filter @image-canvas-editor/editor-core test`、`pnpm --filter @image-canvas-editor/web-vue typecheck`

```ts
import type { EditorState, FilterPreset, ImageResource, Rect } from './types';

export interface HistorySnapshot {
  imageDataUrl: string | null;
  cropRect: Rect | null;
  adjustments: EditorState['adjustments'];
  transform: EditorState['transform'];
  activePreset: FilterPreset;
}

export const captureHistorySnapshot = (state: EditorState): HistorySnapshot => ({
  imageDataUrl: state.image?.dataUrl ?? null,
  cropRect: state.cropRect,
  adjustments: {
    ...state.adjustments,
  },
  transform: {
    ...state.transform,
  },
  activePreset: state.activePreset,
});

export const snapshotsEqual = (left: HistorySnapshot, right: HistorySnapshot): boolean =>
  left.imageDataUrl === right.imageDataUrl &&
  left.activePreset === right.activePreset &&
  left.transform.rotation === right.transform.rotation &&
  left.transform.flipX === right.transform.flipX &&
  left.transform.flipY === right.transform.flipY &&
  left.adjustments.contrast === right.adjustments.contrast &&
  left.adjustments.exposure === right.adjustments.exposure &&
  left.adjustments.highlights === right.adjustments.highlights &&
  left.cropRect?.x === right.cropRect?.x &&
  left.cropRect?.y === right.cropRect?.y &&
  left.cropRect?.width === right.cropRect?.width &&
  left.cropRect?.height === right.cropRect?.height;

export const pushHistorySnapshot = (
  stack: HistorySnapshot[],
  snapshot: HistorySnapshot,
  limit: number,
): HistorySnapshot[] => {
  const lastSnapshot = stack.at(-1);

  if (lastSnapshot && snapshotsEqual(lastSnapshot, snapshot)) {
    return stack;
  }

  const nextStack = [...stack, snapshot];
  return nextStack.length > limit ? nextStack.slice(nextStack.length - limit) : nextStack;
};

export const applyHistorySnapshot = (
  currentState: EditorState,
  snapshot: HistorySnapshot,
  resolveImage: (imageDataUrl: string | null) => ImageResource | null,
): EditorState => ({
  ...currentState,
  image: resolveImage(snapshot.imageDataUrl),
  cropRect: snapshot.cropRect,
  draftCropRect: null,
  cropMode: false,
  adjustments: {
    ...snapshot.adjustments,
  },
  transform: {
    ...snapshot.transform,
  },
  activePreset: snapshot.activePreset,
});
```

- [ ] **Step 3: 在 `editor.ts` 里接入图片注册表、统一提交流程和 `undo/redo` API**

```ts
import {
  applyHistorySnapshot,
  captureHistorySnapshot,
  pushHistorySnapshot,
  snapshotsEqual,
  type HistorySnapshot,
} from './history';

const HISTORY_LIMIT = 50;

export class ImageCanvasEditor {
  private readonly imageRegistry = new Map<string, ImageResource>();
  private undoStack: HistorySnapshot[] = [];
  private redoStack: HistorySnapshot[] = [];

  canUndo(): boolean {
    return this.undoStack.length > 0;
  }

  canRedo(): boolean {
    return this.redoStack.length > 0;
  }

  undo(): void {
    const previousSnapshot = this.undoStack.at(-1);

    if (!previousSnapshot) {
      return;
    }

    this.undoStack = this.undoStack.slice(0, -1);
    this.redoStack = pushHistorySnapshot(
      this.redoStack,
      captureHistorySnapshot(this.store.getState()),
      HISTORY_LIMIT,
    );
    this.store.setState(
      applyHistorySnapshot(this.store.getState(), previousSnapshot, this.resolveHistoryImage),
    );
    this.render();
  }

  redo(): void {
    const nextSnapshot = this.redoStack.at(-1);

    if (!nextSnapshot) {
      return;
    }

    this.redoStack = this.redoStack.slice(0, -1);
    this.undoStack = pushHistorySnapshot(
      this.undoStack,
      captureHistorySnapshot(this.store.getState()),
      HISTORY_LIMIT,
    );
    this.store.setState(applyHistorySnapshot(this.store.getState(), nextSnapshot, this.resolveHistoryImage));
    this.render();
  }

  private readonly resolveHistoryImage = (imageDataUrl: string | null): ImageResource | null => {
    if (!imageDataUrl) {
      return null;
    }

    const image = this.imageRegistry.get(imageDataUrl);

    if (!image) {
      throw new Error('历史图片资源缺失，无法回放快照');
    }

    return image;
  };

  private rememberImage(image: ImageResource | null): void {
    if (image) {
      this.imageRegistry.set(image.dataUrl, image);
    }
  }

  private clearHistory(): void {
    this.undoStack = [];
    this.redoStack = [];
  }

  private commitChange(updater: Partial<EditorState> | ((currentState: EditorState) => EditorState)): void {
    const currentState = this.store.getState();
    const nextState =
      typeof updater === 'function'
        ? updater(currentState)
        : {
            ...currentState,
            ...updater,
          };
    const currentSnapshot = captureHistorySnapshot(currentState);
    const nextSnapshot = captureHistorySnapshot(nextState);

    this.rememberImage(nextState.image);

    if (snapshotsEqual(currentSnapshot, nextSnapshot)) {
      this.store.setState(nextState);
      this.render();
      return;
    }

    this.undoStack = pushHistorySnapshot(this.undoStack, currentSnapshot, HISTORY_LIMIT);
    this.redoStack = [];
    this.store.setState(nextState);
    this.render();
  }
}
```
